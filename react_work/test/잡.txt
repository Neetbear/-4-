  constructor(props) {
    super(props);
    this.state = {
      title: null
    }
  }

  componentDidMount() {
    fetch('http://localhost:5000/api')
      .then(res => res.json())
      .then(data => this.setState({title: data.title}));
  }

  class App extends React.Component{
  render() {
    return (
      <Router>
        <div className="App">
          <header className="App-header">
            <Link to="/">
              <button>Home</button>
            </Link>
            <Link to="/signup">
              <button>Signup</button>
            </Link>
            <Link to="/signin">
              <button>Signin</button>
            </Link>
          </header>
          <hr />
          <main>
            <Route exact path="/" component={Home} />
            <Route path="/about" component={Signup} />
            <Route path="/users" component={Signin} />
          </main>
        </div>
      </Router>
    );
  }
};
